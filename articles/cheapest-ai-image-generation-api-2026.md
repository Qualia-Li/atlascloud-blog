## The Smart Approach: Multi-Model Routing

The most cost-effective strategy is not picking a single model -- it is routing different requests to different models based on the use case. Here is what that looks like in practice:

```python
import requests
import time

API_KEY = "your_api_key_here"
BASE_URL = "https://api.atlascloud.ai/api/v1"

def select_model(use_case):
    """Route to the cheapest model that meets quality requirements."""
    routing = {
        "thumbnail": "z-ai/z-image-turbo/text-to-image",        # $0.01
        "draft": "z-ai/z-image-turbo/text-to-image",             # $0.01
        "social_media": "bytedance/seedream-v5.0-lite",  # $0.032
        "product_catalog": "bytedance/seedream-v5.0-lite",  # $0.032
        "blog_illustration": "google/nano-banana-2/text-to-image",  # $0.013
        "hero_image": "google/imagen4-ultra/text-to-image",      # $0.054
        "premium_brand": "google/imagen4-ultra/text-to-image",   # $0.054
    }
    return routing.get(use_case, "bytedance/seedream-v5.0-lite")

def generate_image(prompt, use_case, width=1024, height=1024):
    """Generate an image using the most cost-effective model for the use case."""
    model = select_model(use_case)
    response = requests.post(
        f"{BASE_URL}/model/prediction",
        headers={"Authorization": f"Bearer {API_KEY}"},
        json={
            "model": model,
            "input": {
                "prompt": prompt,
                "width": width,
                "height": height
            }
        }
    )
    request_id = response.json()["request_id"]

    while True:
        result = requests.get(
            f"{BASE_URL}/model/prediction/{request_id}/get",
            headers={"Authorization": f"Bearer {API_KEY}"}
        )
        data = result.json()
        if data["status"] == "completed":
            return data["output"]["image_url"]
        elif data["status"] == "failed":
            raise Exception(data["error"])
        time.sleep(2)

# Example usage
thumbnail = generate_image(
    "Modern smartphone on gradient background",
    use_case="thumbnail"
)  # Costs $0.01

hero = generate_image(
    "Photorealistic luxury watch on dark marble, dramatic lighting",
    use_case="hero_image",
    width=2048,
    height=2048
)  # Costs $0.054
```
